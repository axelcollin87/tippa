model User {
  id              String              @id @default(uuid())
  email           String              @unique
  name            String
  password        String              // Hashed password
  isAdmin         Boolean             @default(false)
  isApproved      Boolean             @default(false) // Måste godkännas av Admin
  matchBets       MatchBet[]
  groupPlacements GroupPlacementBet[]
  totalScore      Int                 @default(0) 
  createdAt       DateTime            @default(now())
}

model Match {
  id            String     @id @default(uuid())
  stage         String     
  groupName     String?
  homeTeam      String
  awayTeam      String
  kickoff       DateTime
  ground        String?
  homeScore     Int?       
  awayScore     Int?
  actualSign    String?    
  actualWinner  String?    // Laget som gick vidare (slutspelet)
  isCompleted   Boolean    @default(false)
  bets          MatchBet[]
}

model MatchBet {
  id                    String   @id @default(uuid())
  userId                String
  matchId               String
  predictedSign         String   
  predictedWinner       String?  // För slutspelet (vem går vidare?)
  pointsAwarded         Int      @default(0) 
  pointsAwardedProgress Int      @default(0) // Poäng för avancemangs-tips
  user                  User     @relation(fields: [userId], references: [id])
  match                 Match    @relation(fields: [matchId], references: [id])

  @@unique([userId, matchId])
}

model GroupPlacementBet {
  id            String @id @default(uuid())
  userId        String
  groupName     String 
  teamName      String
  predictedRank Int
  pointsAwarded Int    @default(0) 
  user          User   @relation(fields: [userId], references: [id])

  @@unique([userId, groupName, teamName])
}
